point A = (0, 0, 0)

point B = (2, 0, 0)

point C = (1, 1.732, 0)

point D = (1, 0.577, 1.633)

complex tetrahedron = [A, B, C, D]

function clone_and_shift(target, factor, dx, dy, dz) {
    temp = scale(target, factor)
    shifted_result = translate(temp, (dx, dy, dz))
    return shifted_result
}

complex shifted_tetra = clone_and_shift(tetrahedron, 1.5, 4, 0, 1)

complex tetra_boundary = boundary(tetrahedron)

complex shifted_boundary = clone_and_shift(tetra_boundary, 3, 8, 0, 2)

render tetra_boundary
render shifted_tetra
render shifted_boundary
